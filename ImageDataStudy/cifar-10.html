# --- PHASE 2: DATASET CLASSES ---
class CIFAR10TrainDataset(Dataset):
    """Dataset with labels for training and validation."""
    def __init__(self, csv_file, transform=None):
        data = pd.read_csv(csv_file)
        # Pixels: first 3072 | Label: Last column
        self.images = data.iloc[:, :3072].values.astype('uint8')
        self.labels = data.iloc[:, -1].values 
        self.transform = transform
        if self.labels.min() == 1: self.labels = self.labels - 1
        print(f"Train Source Loaded: {len(self.labels)} samples.")

    def __len__(self): return len(self.labels)
    def __getitem__(self, idx):
        image = self.images[idx].reshape(3, 32, 32).transpose(1, 2, 0)
        label = int(self.labels[idx])
        if self.transform: image = self.transform(image)
        return image, label

class CIFAR10TestDataset(Dataset):
    """Dataset without labels (Inference only)."""
    def __init__(self, csv_file, transform=None):
        data = pd.read_csv(csv_file)
        # No label column, all columns are pixels
        self.images = data.values.astype('uint8')
        self.transform = transform
        print(f"Test Source (No Labels) Loaded: {len(self.images)} samples.")

    def __len__(self): return len(self.images)
    def __getitem__(self, idx):
        image = self.images[idx].reshape(3, 32, 32).transpose(1, 2, 0)
        if self.transform: image = self.transform(image)
        return image, -1 # Returning dummy label -1

# Normalization
transform_cifar = transforms.Compose([
    transforms.ToPILImage(),
    transforms.Resize((IMG_SIZE, IMG_SIZE)),
    transforms.ToTensor(),
    transforms.Normalize((0.5, 0.5, 0.5), (0.5, 0.5, 0.5))
])

# Loading and Splitting
full_dataset = CIFAR10TrainDataset('cifar-10_train.csv', transform=transform_cifar)
inference_dataset = CIFAR10TestDataset('cifar-10_test.csv', transform=transform_cifar)

# CRITICAL: We split the train file to get a labeled validation set
train_size = int(0.8 * len(full_dataset))
val_size = len(full_dataset) - train_size
train_data, val_data = random_split(full_dataset, [train_size, val_size])

train_loader = DataLoader(train_data, batch_size=BATCH_SIZE, shuffle=True)
val_loader = DataLoader(val_data, batch_size=BATCH_SIZE, shuffle=False)
inference_loader = DataLoader(inference_dataset, batch_size=BATCH_SIZE, shuffle=False)

# --- PHASE 3: MODEL & LOSS ---
class CustomCNN(nn.Module):
    def __init__(self):
        super(CustomCNN, self).__init__()
        self.conv1 = nn.Conv2d(3, 32, kernel_size=3, padding=1)
        self.bn1 = nn.BatchNorm2d(32)
        self.conv2 = nn.Conv2d(32, 64, kernel_size=3, padding=1)
        self.bn2 = nn.BatchNorm2d(64)
        self.conv3 = nn.Conv2d(64, 128, kernel_size=3, padding=1)
        self.bn3 = nn.BatchNorm2d(128)
        self.pool = nn.MaxPool2d(2, 2)
        self.fc1 = nn.Linear(128 * 4 * 4, 256)
        self.dropout = nn.Dropout(0.5)
        self.fc2 = nn.Linear(256, 10)

    def forward(self, x):
        x = self.pool(F.relu(self.bn1(self.conv1(x))))
        x = self.pool(F.relu(self.bn2(self.conv2(x))))
        x = self.pool(F.relu(self.bn3(self.conv3(x))))
        x = x.view(x.size(0), -1); x = F.relu(self.fc1(x)); x = self.dropout(x)
        return self.fc2(x)

class FocalLoss(nn.Module):
    def __init__(self, alpha=1, gamma=2):
        super(FocalLoss, self).__init__()
        self.alpha, self.gamma = alpha, gamma
    def forward(self, inputs, targets):
        ce_loss = F.cross_entropy(inputs, targets, reduction='none')
        pt = torch.exp(-ce_loss)
        return (self.alpha * (1 - pt)**self.gamma * ce_loss).mean()

loss_functions = {
    "CrossEntropy": nn.CrossEntropyLoss(),
    "MSE": nn.MSELoss(),
    "LabelSmoothing": nn.CrossEntropyLoss(label_smoothing=0.1),
    "FocalLoss": FocalLoss()
}

# --- PHASE 5: EXECUTION ---
results_history = {}
trained_models = {}

for name, criterion in loss_functions.items():
    print(f"\n--- Training: {name} ---")
    model = CustomCNN().to(DEVICE)
    optimizer = torch.optim.Adam(model.parameters(), lr=LEARNING_RATE)
    history = {'train_loss': [], 'val_acc': []}
    
    for epoch in range(EPOCHS):
        model.train()
        epoch_l = 0
        for imgs, lbls in train_loader:
            imgs, lbls = imgs.to(DEVICE), lbls.to(DEVICE)
            optimizer.zero_grad()
            outs = model(imgs)
            if name == "MSE":
                loss = criterion(F.softmax(outs, dim=1), F.one_hot(lbls, 10).float())
            else:
                loss = criterion(outs, lbls)
            loss.backward(); optimizer.step(); epoch_l += loss.item()
            
        model.eval()
        correct, total = 0, 0
        with torch.no_grad():
            for imgs, lbls in val_loader:
                imgs, lbls = imgs.to(DEVICE), lbls.to(DEVICE)
                outs = model(imgs)
                _, p = torch.max(outs, 1); total += lbls.size(0); correct += (p == lbls).sum().item()
        
        acc = 100 * correct / total
        history['train_loss'].append(epoch_l/len(train_loader))
        history['val_acc'].append(acc)
        print(f"Epoch [{epoch+1}] Loss: {epoch_l/len(train_loader):.4f} | Val Acc: {acc:.5f}%")
        
    results_history[name] = history
    trained_models[name] = model

# --- PHASE 6: FINAL ANALYSIS (Using Validation Set) ---
def finalize_thesis_report(results, models, loader):
    # 1. Plot Learning Curves
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(16, 6))
    for n, h in results.items():
        ax1.plot(h['train_loss'], label=n)
        ax2.plot(h['val_acc'], label=n)
    ax1.set_title('CIFAR-10 Training Loss'); ax1.legend(); ax1.grid(True)
    ax2.set_title('CIFAR-10 Validation Accuracy'); ax2.legend(); ax2.grid(True)
    plt.show()

    # 2. Detailed Performance Metrics (Digits=5)
    cifar_classes = ['Airplane', 'Automobile', 'Bird', 'Cat', 'Deer', 'Dog', 'Frog', 'Horse', 'Ship', 'Truck']
    for n, m in models.items():
        m.eval()
        y_true, y_pred = [], []
        with torch.no_grad():
            for imgs, lbls in loader:
                imgs, lbls = imgs.to(DEVICE), lbls.to(DEVICE)
                outs = m(imgs)
                _, p = torch.max(outs, 1)
                y_true.extend(lbls.cpu().numpy())
                y_pred.extend(p.cpu().numpy())
        
        print(f"\n{'='*60}")
        print(f"DETAILED REPORT FOR: {n} (Evaluated on Validation Set)")
        print(f"{'='*60}")
        print(classification_report(y_true, y_pred, target_names=cifar_classes, digits=5))

        # 3. Confusion Matrix for Error Analysis
        plt.figure(figsize=(8, 6))
        cm = confusion_matrix(y_true, y_pred)
        sns.heatmap(cm, annot=True, fmt='d', cmap='Oranges', xticklabels=cifar_classes, yticklabels=cifar_classes)
        plt.title(f'Confusion Matrix: {n}')
        plt.ylabel('Actual Label')
        plt.xlabel('Predicted Label')
        plt.show()

# --- CORRECTED EXECUTION ---
# We use val_loader here because test_loader (inference) has no labels!
finalize_thesis_report(results_history, trained_models, val_loader)
